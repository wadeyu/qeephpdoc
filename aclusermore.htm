/*

  • --acl_discuz_key 为用户表中的随即加密字符串,如果此设置为空则不使用此方式加密
  • -取消登录方法中无用的已登录判断

class Model_Behavior_AclUserMore extends Model_Behavior_AclUser

{

function IsLogin()

{

$key = Q::ini(‚acl_session_key‘) ? Q::ini(‚acl_session_key‘) : ‚user‘;

$cookieKey = Q::ini(‚session_cookie_name‘) ? Q::ini(‚session_cookie_name‘) : ‚user‘;

if(!empty($_SESSION[$key]))

return 1;

else if(isset($_COOKIE[$cookieKey]))

{

return $this->LoginToWebSite($_COOKIE[$cookieKey][‚username‘], $_COOKIE[$cookieKey][‚password‘],NULL,true);

}

else

return –1;

}

/*

* 登录到网站 产生SESSION 和 COOKIES 返回操作结果状态

*

* 如果登录成功SESSION自动产生 $key为 environment.yaml.php 中的acl_session_key

* -如果acl_session_key为空 则 为默认值‚user‘

* -产生SESSION为$_SESSION[$key][‚uid‘]和$_SESSION[$key][‚username‘]

*

* 如果$loginTime>0则产生COOKIE 反之不产生 cookie

* -$cookieKey为 environment.yaml.php 中的session_cookie_name

* -如果session_cookie_name为空 则 为默认值‚user‘

* -产出 cookie为 $_COOKIE[$cookieKey][‚uid‘],$_COOKIE[$cookieKey][‚username‘]

* -和 $_COOKIE[$cookieKey][‚password‘]

* -cookies path 为 environment.yaml.php 中的acl_cookie_path

* -如果acl_cookie_path为空 则 为默认值‚/‘

* 如果 acl_cookie_salt 为空则不使用 MD5再次加密

*

* 用法：

* @code php

*

* //SESSION失效 但COOKIES未失效

* $this->LoginToWebSite($_COOKIE[‚user‘][‚username‘],$_COOKIE[‚user‘][‚password‘],NULL,true);

*

* //登录 产生SESSION和COOKIE $time为cookie有效期

* $this->LoginToWebSite($_COOKIE[‚user‘][‚username‘],$_COOKIE[‚user‘][‚password‘],$time);

*

* //登录 只产生SESSION

* $this->LoginToWebSite($_COOKIE[‚user‘][‚username‘],$_COOKIE[‚user‘][‚password‘]);

* @endcode

*

* 如果用户名不存在，将抛出 AclUser_UsernameNotFoundException 异常。

* 密码不正确，则抛出 AclUser_WrongPasswordException 异常。

*

* @param string $username 用户名

* @param string $password 密码

* @param int $loginTime cookies 有效期

* @$cookieslogin 是否cookies自动登录产生SESSION

*

*/

function LoginToWebSite($username, $password, $loginTime = 0, $cookieslogin = false)

{

$key = Q::ini(‚acl_session_key‘) ? Q::ini(‚acl_session_key‘) : ‚user‘;

$cookieKey = Q::ini(‚session_cookie_name‘) ? Q::ini(‚session_cookie_name‘) : ‚user‘;

if(Q::ini(‚acl_discuz_key‘))

{

$pn = $this->_settings[‚username_prop‘];

$member = $this->_meta->find(array($pn ⇒ $username))->setColumns(Q::ini(‚acl_discuz_key‘))->query();

$salt = $member[Q::ini(‚acl_discuz_key‘)];

}

else

$salt = Q::ini(‚acl_cookie_salt‘) ? Q::ini(‚acl_cookie_salt‘) : '';

/*

if(isset($_SESSION[$key]))

;return –3;已经登录了

*/

if($cookieslogin)

$objectUser = $this->validateCookieLogin($username, $password);

else

{

if(!empty($salt))

$password = md5($password).$salt;

$objectUser = $this->validateLogin($username, $password);

}

//登录成功处理

$_SESSION[$key][‚uid‘] = $objectUser->uid;

$_SESSION[$key][‚username‘] = $username;

if(!empty($loginTime))

{

$path = Q::ini(‚acl_cookie_path‘) ? Q::ini(‚acl_cookie_path‘) : ‚/‘;

setcookie($cookieKey.‚[uid]‘, $objectUser->uid, time()+$loginTime, $path);

setcookie($cookieKey.‚[username]‘, $objectUser->username, time()+$loginTime, $path);

setcookie($cookieKey.‚[password]‘, $objectUser->password, time()+$loginTime, $path);

}

return $objectUser;//登录成功

}

/*

* 与validateLogin类似

* 但不再将获得的$password进行加密

* 此方法并非原创 来源:: QEEphp.com :aligo 发表于 2009–01–15 15:47

*/

function validateCookieLogin($username, $password, $update_login = true)

{

$pn = $this->_settings[‚username_prop‘];

$member = $this->_meta->find(array($pn ⇒ $username))->query();

if (!$member->id())

{

throw new AclUser_UsernameNotFoundException($username);

}

if ($member[$this->_settings[‚password_prop‘]] != $password)

{

throw new AclUser_WrongPasswordException($username);

}

if ($this->_settings[‚update_login_auto‘])

{

$this->updateLoginDyn($member);

}

return $member;

}

/*

*退出登录状态

*删除 SESSION 和COOKIE

*

*/

function OutWebSite()

{

$key = Q::ini(‚acl_session_key‘) ? Q::ini(‚acl_session_key‘) : ‚user‘;

$cookieKey = Q::ini(‚session_cookie_name‘) ? Q::ini(‚session_cookie_name‘) : ‚user‘;

//注销用户

unset($_SESSION[$key]);

//注销用户cookies

setcookie($cookieKey.‚[uid]‘,0, time()-1, Q::ini(‚acl_cookie_path‘));

setcookie($cookieKey.‚[username]‘ ,0, time()-1, Q::ini(‚acl_cookie_path‘));

setcookie($cookieKey.‚[password]‘ ,0, time()-1, Q::ini(‚acl_cookie_path‘));

}

}
